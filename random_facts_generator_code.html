            import tkinter as tk
            from tkinter import *
            import randfacts
            
            
            # Function to add facts
            def move(event=None):
                if event:
                    print(f"Event type: {event.type}")
                facts = randfacts.get_fact(True)
                c = f"{len(text_area.get('1.0', 'end').splitlines())}) "  # Get the current line count
                text_area.insert(tk.END, c + facts + "\n")
                text_area.see(tk.END)  # Scroll to the bottom after adding a new fact
            
            
            # Function to clear the text area
            def clear():
                text_area.delete('1.0', tk.END)
            
            
            # Function to close the window
            def destroy():
                root.destroy()
            
            
            # Driver code
            root = tk.Tk()
            root.geometry("1200x400")
            
            # Add space between the top and the "Get Fact" button
            tk.Label(root).pack(pady=10)
            
            # Create a scrollable text area
            scrollbar = tk.Scrollbar(root)
            scrollbar.pack(side=tk.RIGHT, fill=tk.Y)
            
            text_area = tk.Text(root, yscrollcommand=scrollbar.set)
            text_area.pack(side=tk.LEFT, fill=tk.BOTH, expand=True)
            
            scrollbar.config(command=text_area.yview)
            
            # Add buttons
            button = tk.Button(root, text="Click here for Facts", command=move)
            button.pack()
            
            button_frame = tk.Frame(root)
            button_frame.pack()
            
            button_clear = tk.Button(button_frame, text="Clear", command=clear)
            button_clear.pack(pady=5)
            
            button_quit = tk.Button(button_frame, text="Quit", command=destroy)
            button_quit.pack(pady=5)
            
            root.mainloop()
